-- ==============================================
-- FUNGSI SCAN SEMUA REMOTEEVENT
-- ==============================================
local function scanAllRemotes()
    notify("üîç Scanning for RemoteEvents...")
    remoteEvents = {}
    
    local function scanContainer(container)
        for _, obj in ipairs(container:GetDescendants()) do
            if obj:IsA("RemoteEvent") then
                table.insert(remoteEvents, {
                    obj = obj,
                    name = obj.Name,
                    path = obj:GetFullName(),
                    parent = obj.Parent.Name
                })
            end
        end
    end
    
    scanContainer(ReplicatedStorage)
    pcall(function() scanContainer(workspace) end)
    pcall(function() scanContainer(game:GetService("Lighting")) end)
    
    notify("‚úÖ Found " .. #remoteEvents .. " RemoteEvents")
    return remoteEvents
end

        {["cmd"] = "bc", ["msg"] = broadcastMessage},

        broadcastMessage,  -- Plain string

    return broadcastPayloads[math.random(1, #broadcastPayloads)]
end

-- ==============================================
-- FUNGSI SPAM REMOTE (DENGAN COUNT)
-- ==============================================
local function spamRemote(remote, count)
    if not remote or not remote:IsA("RemoteEvent") then
        notify("‚ùå Invalid remote")
        return
    end
    
    notify("üöÄ Spamming: " .. remote.Name)
    
    local thread = coroutine.create(function()
        local spammed = 0
        local maxSpam = count or config.spamCount
        
        while isSpamming and spammed < maxSpam do
            pcall(function()
                if broadcastEnabled then
                    -- Broadcast mode - send message in various formats
                    remote:FireServer(generateBroadcastPayload())
                    remote:FireServer("announce", broadcastMessage)
                    remote:FireServer("broadcast", broadcastMessage)
                    remote:FireServer("chat", broadcastMessage)
                    remote:FireServer("say", broadcastMessage)
                else
                    -- Normal spam mode
                    remote:FireServer(generatePayload())
                    remote:FireServer({cmd = "execute", code = generatePayload()})
                    remote:FireServer("admin", player.Name, true)
                end
            end)
            
            spammed = spammed + 3
            spamCount = spamCount + 3
            
            task.wait(config.spamDelay)
        end
    end)
    
    table.insert(spamThreads, thread)
    coroutine.resume(thread)
end

-- ==============================================
-- FUNGSI SPAM ALL REMOTES
-- ==============================================
local function spamAllRemotes()
    if #remoteEvents == 0 then
        scanAllRemotes()
    end
    
    if #remoteEvents == 0 then
        notify("‚ùå No RemoteEvents found")
        return
    end
    
    isSpamming = true
    spamThreads = {}
    
    local mode = broadcastEnabled and "BROADCAST" or "SPAM"
    notify("üî• " .. mode .. " " .. #remoteEvents .. " RemoteEvents...")
    
    for i, remote in ipairs(remoteEvents) do
        local thread = coroutine.create(function()
            local spammed = 0
            while isSpamming and spammed < config.spamCount do
                pcall(function()
                    if broadcastEnabled then
                        remote.obj:FireServer(generateBroadcastPayload())
                        remote.obj:FireServer("broadcast", broadcastMessage)
                    else
                        remote.obj:FireServer(generatePayload())
                        remote.obj:FireServer("execute", generatePayload())
                    end
                    spammed = spammed + 2
                end)
                task.wait(config.spamDelay / 2)
            end
        end)
        
        table.insert(spamThreads, thread)
        coroutine.resume(thread)
        task.wait(0.01)
    end
end

-- ==============================================
-- FUNGSI STOP SPAM
-- ==============================================
local function stopSpam()
    isSpamming = false
    spamThreads = {}

        pcall(function() conn:Disconnect() end)
    end
    spamConnections = {}
    notify("‚èπÔ∏è Stopped all spam")
end
